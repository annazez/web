/**
 * THEME BOOTSTRAP:
 * This script must run synchronously before paint to prevent FOUC.
 * CSP: If this script is modified, the SHA-256 hash in Content-Security-Policy must be updated.
 */
const root = document.documentElement;

const getPreferredTheme = () => {
  const storedTheme =
    typeof localStorage !== 'undefined' ? localStorage.getItem('theme') : null;
  if (storedTheme === 'dark' || storedTheme === 'light') {
    return storedTheme;
  }
  return window.matchMedia('(prefers-color-scheme: dark)').matches ? 'dark' : 'light';
};

const applyPreferredTheme = () => {
  const theme = getPreferredTheme();
  if (theme === 'dark') {
    root.classList.add('dark');
    root.style.colorScheme = 'dark';
  } else {
    root.classList.remove('dark');
    root.style.colorScheme = 'light';
  }
};

applyPreferredTheme();
window.__applyPreferredTheme = applyPreferredTheme;
